    <script>
        // Điền lại chính xác ID file Google Sheets của bạn vào đây
        const SHEET_ID = '1tAhUvW__1s77kGjRyejXmg3lKPQ_H7SoWswmBGrrVn8'; 
        
        // Đường link lấy dữ liệu dạng CSV (Bảo mật, mượt mà và không bao giờ lỗi 404)
        const url = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/export?format=csv&gid=0`;

        fetch(url)
            .then(res => res.text())
            .then(csvText => {
                // Hàm tách dữ liệu CSV thành mảng dữ liệu sạch
                const lines = csvText.split("\n").map(line => {
                    // Xử lý dấu phẩy và dấu ngoặc kép trong file CSV
                    return line.match(/(".*?"|[^",\s]+)(?=\s*,|\s*$)/g) || [];
                });
                
                let html = '';
                
                // Lặp qua các dòng dữ liệu (Bỏ qua dòng tiêu đề thứ 1)
                for (let i = 1; i < lines.length; i++) {
                    const row = lines[i];
                    if (row.length < 2) continue; // Bỏ qua dòng trống

                    // Hàm xóa dấu ngoặc kép thừa ở đầu và cuối chữ
                    const clean = (str) => str ? str.replace(/^"|"$/g, '').trim() : '';

                    const id = clean(row[0]);
                    const title = clean(row[1]);
                    const category = clean(row[2]);
                    const description = clean(row[3]);
                    const image = clean(row[4]);
                    const price = clean(row[5]);
                    const link = clean(row[6]);

                    let buttonHTML = '';
                    if (price === 'Miễn phí' || price === '0' || price === '') {
                        buttonHTML = `<a href="${link}" target="_blank" class="w-full block text-center bg-green-500 hover:bg-green-600 text-white font-medium py-2 rounded-md transition">Tải Miễn Phí</a>`;
                    } else {
                        buttonHTML = `<a href="${link}" target="_blank" class="w-full block text-center bg-blue-600 hover:bg-blue-700 text-white font-medium py-2 rounded-md transition">Mua Ngay (${price})</a>`;
                    }

                    html += `
                        <div class="bg-white rounded-xl shadow-sm border border-gray-100 overflow-hidden hover:shadow-md transition">
                            <img src="${image}" alt="${title}" class="w-full h-48 object-cover">
                            <div class="p-5">
                                <span class="text-xs font-semibold uppercase tracking-wider px-2 py-1 bg-red-50 text-red-600 rounded">
                                    ${category}
                                </span>
                                <h3 class="font-bold text-lg text-gray-900 mt-3 line-clamp-2">${title}</h3>
                                <p class="text-gray-500 text-sm mt-1 mb-4 line-clamp-2">${description}</p>
                                ${buttonHTML}
                            </div>
                        </div>
                    `;
                }

                document.getElementById('danh-sach-tai-lieu').innerHTML = html;
            })
            .catch(err => {
                document.getElementById('danh-sach-tai-lieu').innerHTML = '<p class="text-center col-span-3 text-red-500">Lỗi hệ thống! Vui lòng kiểm tra lại kết nối mạng hoặc file Google Sheets.</p>';
                console.error(err);
            });
    </script>
