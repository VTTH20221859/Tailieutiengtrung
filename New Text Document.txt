    <script>
        // CHÚ Ý: Hãy thay đoạn chữ bên dưới bằng ID file Google Sheets bạn copy ở Bước 1
        const SHEET_ID = '1tAhUvW__1s77kGjRyejXmg3lKPQ_H7SoWswmBGrrVn8';
        
       const url = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:json&sheet=Sheet1`;

        fetch(url)
            .then(res => res.text())
            .then(data => {
                // Định dạng lại dữ liệu nhận về từ Google
                const temp = data.substring(47, data.length - 2);
                const json = JSON.parse(temp);
                const rows = json.table.rows;
                
                let html = '';
                
                // Đọc từng dòng dữ liệu và vẽ giao diện
                rows.forEach(row => {
                    const id = row.c[0] ? row.c[0].v : '';
                    const title = row.c[1] ? row.c[1].v : 'Chưa có tên';
                    const category = row.c[2] ? row.c[2].v : 'Chưa phân mục';
                    const description = row.c[3] ? row.c[3].v : '';
                    const image = row.c[4] ? row.c[4].v : 'https://via.placeholder.com/150';
                    const price = row.c[5] ? row.c[5].v : 'Miễn phí';
                    const link = row.c[6] ? row.c[6].v : '#';

                    // Kiểm tra nút bấm là Tải miễn phí hay Mua qua Chatbot
                    let buttonHTML = '';
                    if (price === 'Miễn phí' || price === 0) {
                        buttonHTML = `<a href="${link}" target="_blank" class="w-full block text-center bg-green-500 hover:bg-green-600 text-white font-medium py-2 rounded-md transition">Tải Miễn Phí</a>`;
                    } else {
                        buttonHTML = `<a href="${link}" target="_blank" class="w-full block text-center bg-blue-600 hover:bg-blue-700 text-white font-medium py-2 rounded-md transition">Mua Ngay (${price})</a>`;
                    }

                    html += `
                        <div class="bg-white rounded-xl shadow-sm border border-gray-100 overflow-hidden hover:shadow-md transition">
                            <img src="${image}" alt="${title}" class="w-full h-48 object-cover">
                            <div class="p-5">
                                <span class="text-xs font-semibold uppercase tracking-wider px-2 py-1 bg-red-50 text-red-600 rounded">
                                    ${category}
                                </span>
                                <h3 class="font-bold text-lg text-gray-900 mt-3 line-clamp-2">${title}</h3>
                                <p class="text-gray-500 text-sm mt-1 mb-4 line-clamp-2">${description}</p>
                                ${buttonHTML}
                            </div>
                        </div>
                    `;
                });

                document.getElementById('danh-sach-tai-lieu').innerHTML = html;
            })
            .catch(err => {
                document.getElementById('danh-sach-tai-lieu').innerHTML = '<p class="text-center col-span-3 text-red-500">Lỗi không thể tải dữ liệu. Vui lòng kiểm tra lại ID Google Sheets!</p>';
                console.error(err);
            });
    </script>
